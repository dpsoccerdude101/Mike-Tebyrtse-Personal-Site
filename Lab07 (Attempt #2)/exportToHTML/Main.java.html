121      private static ArrayList<String> getUserDataFile () throws IOException {
122          URL users = new URL("https://raw.githubusercontent.com/dpsoccerdude101/dpsoccerdude101.github.io/master/Lab07%20(Attempt%20%232)/Users.txt");
123          BufferedReader input = new BufferedReader(new InputStreamReader(users.openStream()));
124          ArrayList<String> list = new ArrayList<>();
125          String inputLine;
126          while ((inputLine = input.readLine()) != null) {
127              list.add(inputLine);
128          }
129          input.close();
130          return list;
131      }
132      private static boolean contains (User thatUser) {
133          for (int count = 0; count < usersList.size(); count++) {
134              User tempUser = usersList.get(count);
135              if (tempUser.getUsername().equals(thatUser.getUsername())) {
136                  if (tempUser.getPassword().equals(thatUser.getPassword())) {
137                      return true;
138                  }
139              }
140          }
141          return false;
142      }

144      public class JavaConnector {
145  
146          private String value;
147          /**
148           * called when the JS side wants a String to be read.
149           *
150           * @param value
151           *         the String to read
152           */
153          public void toJavaData(String value) {
154              this.value = value;
155              String[] tokens = value.split("&");
156              Double[] tokensDouble = new Double[tokens.length];
157              for (int count = 0; count < tokens.length; count++) {
158                  String[] miniTokens = tokens[count].split("=");
159                  tokensDouble[count] = Double.parseDouble(miniTokens[1]);
160              }
161              invoice = new Invoice(tokensDouble[0], tokensDouble[1], tokensDouble[2], tokensDouble[3], tokensDouble[4]);
162  
163              if (this.value != null) {
164                  javascriptConnector.call("showResult", ("$" + String.format("%.2f", invoice.getTotalBill())));
165              }
166          }
167          public void toJavaLogin(String value) {
168              this.value = value;
169              String[] tokens = value.split("&");
170              Pair<String, String> pair = new Pair<>((tokens[0].split("="))[1], (tokens[1].split("="))[1]);
171              user = new User(pair.getKey(), pair.getValue());
172  
173              if (contains(user)) {
174                  javascriptConnector.call("goToQueryPage");
175              }
176              else {
177                  javascriptConnector.call("loginFailed");
178              }
179          }
180      }
