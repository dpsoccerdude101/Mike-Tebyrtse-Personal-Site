4    /**
5     * House Model Object Represents a house that is for sale. Defined by the
6     * following data: address, price, area, and number of bedrooms Address: this is
7     * the street address such as 23-Linden-Avenue. Even though there can be many
8     * words, we make it one word by introducing hyphens. Price: this is a number
9     * like 149999. Area: the square-footage of the house, again a number, such as
10    * 1800. numberOfBedrooms: gain, a small number, like 3
11    *
12    * @author Dennis Pavlyuk
13    * @version 1.1
14    */
15   public class House {
16       private String address;
17       private int price;
18       private int area;
19       private int numberOfBedrooms;
20   
21       /**
22        * Constructor.
23        *
24        * @param address
25        * @param price            Integer
26        * @param area             square feet
27        * @param numberOfBedrooms Integer
28        */
29       public House(String address, int price, int area, int numberOfBedrooms) {
30           this.address = address;
31           this.price = price;
32           this.area = area;
33           this.numberOfBedrooms = numberOfBedrooms;
34       }
35   
36       /**
37        * @return the address
38        */
39       public String getAddress() {
40           return address;
41       }
42   
43       /**
44        * @return the price
45        */
46       public int getPrice() {
47           return price;
48       }
49   
50       /**
51        * @return the area
52        */
53       public int getArea() {
54           return area;
55       }
56   
57       /**
58        * @return the numberOfBedrooms
59        */
60       public int getNumberOfBedrooms() {
61           return numberOfBedrooms;
62       }
63   
64       /**
65        * @return string concatenation of house object variables.
66        */
67       public String toString() {
68           String house;
69           house = ("The property at " + address);
70           house += (" costs $" + price + ", has " + area + " square feet ");
71           house += ("and " + numberOfBedrooms + " bedrooms.");
72           return house;
73       }
74       /**
75        * Checks if the callee of type (House) matches the specifications of called Criteria, object c
76        *
77        * @param c (Criteria object)
78        * @return true if this House is within specified criteria, else false.
79        */
80       public boolean satisfies(Criteria c) {
81           if (this.getPrice() >= c.getMinimumPrice()) {
82               if (this.getPrice() <= c.getMaximumPrice()) {
83                   if (this.getArea() >= c.getMinimumArea()) {
84                       if (this.getArea() <= c.getMaximumArea()) {
85                           if (this.getNumberOfBedrooms() >= c.getMinimumNumberOfBedrooms()) {
86                               if (this.getNumberOfBedrooms() <= c.getMaximumNumberOfBedrooms()) {
87                                   return true;
88                               }
89                           }
90                       }
91                   }
92               }
93           }
94           return false;
95       }
