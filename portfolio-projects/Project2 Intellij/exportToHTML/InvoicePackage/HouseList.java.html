11   import java.io.File;
12   import java.io.FileNotFoundException;
13   import java.util.ArrayList;
14   import java.util.Scanner;
15   
16   public class HouseList {
17       private ArrayList<House> houseList = new ArrayList<House>();
18   
19       /**
20        * Constructor.
21        * Reads a file of text creates a house object from it. Then adds that house object to a list of House objects.
22        *
23        * @param fileName
24        */
25       public HouseList(String fileName) {
26           File file = new File(fileName);
27           Scanner fileInput = null;
28           try {
29               fileInput = new Scanner(file);
30           } catch (FileNotFoundException e) {
31               // TODO Auto-generated catch block
32               e.printStackTrace();
33           }
34           while (fileInput.hasNextLine()) {
35               // reads next line of data
36               String str = fileInput.nextLine();
37   
38               // Separate into tokens
39               String[] split = str.split("\\s+");
40               String address = split[0];
41   
42               // format address
43               String[] addressSplit = address.split("-");
44               addressSplit[1] = addressSplit[1].substring(0, 1).toUpperCase() + addressSplit[1].substring(1);
45               addressSplit[2] = addressSplit[2].substring(0, 1).toUpperCase() + addressSplit[2].substring(1);
46               address = String.join(" ", addressSplit);
47               int price = Integer.parseInt(split[1]);
48               int area = Integer.parseInt(split[2]);
49               int numberOfBedrooms = Integer.parseInt(split[3]);
50               House houseObjectFromDataSet = new House(address, price, area, numberOfBedrooms);
51               houseList.add(houseObjectFromDataSet);
52           }
53       }
54          /**
55        * This method uses the House method, satisifies, to see if each House object
56        * within the ArrayList, houseList, matches with Criteria object c and uses the House.toString()
57        * method to print out the concatenated string of the matched house.
58        *
59        * @param c (Criteria object)
60        */
61       public void printHouses(Criteria c) {
62           for (int count = 0; count < this.houseList.size(); count++) {
63               if(this.houseList.get(count).satisfies(c)) {
64                   System.out.println(this.houseList.get(count).toString());
65               }
66           }
67       }
68       public ArrayList<House> getHouses(Criteria c) {
69           ArrayList<House> matchedHouses = new ArrayList<House>();
70           for (int i = 0; i < houseList.size(); i++) {
71               if (houseList.get(i).satisfies(c)) {
72                   matchedHouses.add(houseList.get(i));
73               }
74           }
75           return matchedHouses;
76       }
77       //randomly picks a house in the matched arrayList
78       public House getRandom(ArrayList<House> matchedHouses) {
79   
80           if (matchedHouses.get(0).getAddress().equals("0")) {
81               return new House ("1", 0, 0, 0);
82           }
83           else {
84               int randomIndex = (int) (Math.random() * matchedHouses.size());
85               House matched = matchedHouses.get(randomIndex);
86               return matched;
87           }
88       }
